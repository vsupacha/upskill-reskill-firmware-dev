  • การประกาศโครงสร้าง

  • การอ้างอิงกลุ่มรีจิสเตอร์
typedef struct {
  __IOM uint32_t MODER;     // GPIO port mode register, offset: 0x00
  __IOM uint32_t OTYPER;    // GPIO port output type register, offset: 0x04
  __IOM uint32_t OSPEEDR;   // GPIO port output speed register, offset: 0x08
  __IOM uint32_t PUPDR;     // GPIO port pull-up/pull-down register, offset: 0x0C
  __IM  uint32_t IDR;       // GPIO port input data register, offset: 0x10
  __IOM uint32_t ODR;       // GPIO port output data register, offset: 0x14
  __OM  uint32_t BSRR;      // GPIO port bit set/reset  register, offset: 0x18
  __IOM uint32_t LCKR;      // GPIO port configuration lock register, offset: 0x1C
  __IOM uint32_t AFR[2];    // GPIO alternate function registers, offset: 0x20-0x24
  __OM  uint32_t BRR;       // GPIO port bit Reset register, offset: 0x28
} GPIO_TypeDef;

3. Embedded C programming

  • ตัวกระทำระดับบิต
Operation C code คำอธิบาย
Bit mask (1 << 3) | (1 << 5) สร้าง bit 3 และ 5 เป็น 1 โดย bit อื่นเป็น 0
Set bit REG |= (1 << 5) bit 5 ของ REG เป็น 1 โดย bit อื่นคงเดิม
Clear bit REG &= ~(1 << 5) bit 5 ของ REG เป็น 0 โดย bit อื่นคงเดิม
Toggle bit REG ^= (1 << 5) bit 5 ของ REG สลับสถานะ (1 → 0 หรือ 0 → 1) โดย bit อื่นคงเดิม
Test bit if ( REG & (1 << 5) ) หาก bit 5 ของ REG เป็น 1 ผลคือ True หากเป็น 0 ผลคือ False
Extract bits BYTE = (REG >> 8) & 0xFF แยก bit 8-15 จาก REG

3. Practice: bare-metal coding

  1. ตั้งค่า RCC เพื่อเปิดใช้งานพอร์ท A และ C
RCC->AHB2ENR |= ((1 << 0) | (1 << 2));
  1. ตั้งค่า GPIOA เพื่อกำหนดขา PA5 เป็น OUTPUT
GPIOA->MODER &= ~(1 << (5*2)+1); 
GPIOA->MODER |= (1 << 5*2);
  1. ตั้งค่า GPIOC เพื่อกำหนดขา PC13 เป็น INPUT
GPIOC->MODER &= ~((1 << (13*2)+1) | (1 << (13*2)));
  1. ทดสอบขา PC13 เพื่อควบคุมขา PA5
if (GPIOC->IDR & (1 << 13)) {
    GPIOA->ODR |= (1 << 5);
} else {
    GPIOA->ODR &= ~(1 << 5);
}

3. Firmware behaviour

UARTx_TDR (Tx data register) มีขนาด 1 ไบต์: การส่งข้อความ “hello” ทำอย่างไร

Polling: HAL_UART_Transmit()

  • ย้ายข้อมูล 1 ไบต์เข้า TDR → วนรอบเช็คแฟลก TXE
  • ไม่สามารถประมวลผลโค้ดส่วนอื่น
  • ต้องกำหนดค่า timeout

Interrupt: HAL_UART_Transmit_IT()

  • ตั้ง UART interrupt flag TXEIE ใน USARTx_CR1
  • ย้ายข้อมูล 1 ไบต์เข้า TDR → ประมวลผลโค้ดต่อไป
  • เมื่อเกิดการร้องขอการขัดจังหวะ1 เนื่องจากแฟลก TXE → ย้ายข้อมูล 1 ไบต์ถัดมา

DMA: HAL_UART_Transmit_DMA()

  • ตั้งการทำงานของ DMA: ตำแหน่งหน่วยความจำต้นทาง + ขนาดข้อมูล และตำแหน่งของ UARTx_TDR
  • DMA จะย้ายข้อมูลเมื่อแฟลก TXE มีค่าเป็น 1
  • ไม่มีการขัดจังหวะ ยกเว้นกำหนด DMA IRQ (โอนย้ายข้อมูลครบแล้ว)

3. Embedded programming issues

ปัญหาจาก Interrupt Request

  • ขัดจังหวะระหว่างอ่าน/เขียนข้อมูล → race condition1 หรือ read-modify-write2
int main() {
    count += 1; // เกิด IRQ ระหว่างกลาง

void isr_handler() {
    count += 1;
}
  • วิธีป้องกัน → สร้าง critical section โดยปิดการขัดจังหวะ
int main() {
    __disable_irq();
    count += 1; // เข้าสู่ atomic access
    __enable_irq();
}
